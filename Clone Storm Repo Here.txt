@staticmethod
    def write_str(s, file_path):
        """
        Write string to file.
        """
        os.makedirs(os.path.dirname(os.path.abspath(file_path)), exist_ok=True)
        with open(file_path, "w", encoding="utf-8") as f:  # Explicitly use UTF-8
            f.write(s)
